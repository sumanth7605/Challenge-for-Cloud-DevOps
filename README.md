# Install ansible

sudo yum update -y

sudo yum install python3 python3-pip -y

sudo yum -y install https://dl.fedoraproject.org/pub/epel/epel-release-latest-8.noarch.rpm

sudo yum install  --enablerepo epel-playground  ansible

ansible --version
